query RetrieveFullCart ($cartId: String) {
  cart(id: $cartId) {
    data {
      id
      travelers {
        names {
          firstName
          lastName
        }
      }
      contacts {
        contactType
        ... on Email {
          address
        }
        ... on Phone {
          deviceType
          countryPhoneExtension
          number
        }
      }
      frequentFlyerCards {
        companyCode
        cardNumber
              travelerId
      }
      airOffers {
        id
        offerItems {
          air {
            bounds {
              originLocationCode
              destinationLocationCode
            }
          }
        }
      }
      accommodations {
        code
      }
    }
    dictionaries {
      flight {
        aircraftCode
      }
      location {
        code
      }
    }
  }
}
