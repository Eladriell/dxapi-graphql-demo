query RetrieveFullCart ($cartId: String) {
  cart(id: $cartId) {
      id
      weather {
        date
        mintempC
        maxtempC
        hourly(time: ["1200", "1500"]) {
          time
          tempC
          visibility
          weatherIconUrl
          weatherDesc
        }
      }
      travelers {
        names {
          firstName
          lastName
        }
      }
      contacts {
        contactType
        ... on Email {
          address
        }
        ... on Phone {
          deviceType
          countryPhoneExtension
          number
        }
      }
      frequentFlyerCards {
        companyCode
        cardNumber
              travelerId
      }
      airOffers {
        id
        offerItems {
          air {
            bounds {
              originLocationCode
              destinationLocationCode
            }
          }
        }
      }
      accommodations {
        code
      }
      extensions {
        id
        content
      }
    dictionaries {
      flight {
        aircraftCode
      }
      location {
        code
        cityName
      }
    }
    }
}
